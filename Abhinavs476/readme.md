#include <iostream>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int a[n],sum=0,sumk=0;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
            sum=sum+a[i];
        }
        for(int j=0;j<n;j++)
        {
            if(a[j]>k)
            {
                a[j]=k;
            }
            sumk=sumk+a[j];
        }
        cout<<sum-sumk<<endl;
    }
	return 0;
}
